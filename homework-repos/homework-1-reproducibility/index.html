# Read the HTML file
html_content <- read_html("USCODE23_LLCP_021924.HTML")

# Extract all tables from the HTML
tables <- html_content %>% html_table(fill = TRUE)

# Look for the table containing _AGEG5YR
ageg5yr_table <- NULL
for(i in 1:length(tables)) {
  if(any(grepl("_AGEG5YR|AGEG5YR", tables[[i]], ignore.case = TRUE))) {
    ageg5yr_table <- tables[[i]]
    break
  }
}
